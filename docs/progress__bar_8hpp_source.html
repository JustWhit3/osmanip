612 
613  iterating_var_ = 100 * (iterating_var - min_) / (max_ - min_ - osm::one(iterating_var)),
614  iterating_var_spin_ =
615  osm::isFloatingPoint(iterating_var) ? (osm::roundoff(iterating_var, 1) * 10) : iterating_var,
616  width_ = (iterating_var_ + 1) / 4;
617 
618  // Update of the progress indicator only:
619  if (styles_map_.at("indicator").find(style_) != styles_map_.at("indicator").end()) {
620  output_ = feat(crs, "left", 100) + getColor() +
621  std::to_string(static_cast<int32_t>(round(iterating_var_++))) + feat(rst, "color") +
622  getStyle();
623

628  else if (styles_map_.at("loader").find(style_) != styles_map_.at("loader").end()) {
629  output_ =

719  void remaining_time() {

721 
722  duration time_taken = steady_clock::now() - begin_timer;
723  float percentage_done = static_cast<float>(ticks_occurred) / (max_spin_);
724  duration time_left = time_taken * (1 / percentage_done - 1);
725  std::chrono::minutes minutes_left = std::chrono::duration_cast<std::chrono::minutes>(time_left);
726  std::chrono::seconds seconds_left =
727  std::chrono::duration_cast<std::chrono::seconds>(time_left - minutes_left);
728 
729  osm::cout << "["

731  "Estimated time "
732  "left: " +
733  feat(rst, "italics")
734  << feat(col, "green") << minutes_left.count() << feat(rst, "color") << "m "
