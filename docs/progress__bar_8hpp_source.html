617 
618  iterating_var_ = 100 * (iterating_var - min_) / (max_ - min_ - agr::one(iterating_var)),
619  iterating_var_spin_ =
620  agr::isFloatingPoint(iterating_var) ? (agr::roundoff(iterating_var, 1) * 10) : iterating_var,
621  width_ = (iterating_var_ + 1) / 4;
622

624  // indicator only:

628  getStyle();
629 
630  update_output(output_);

727  void remaining_time() {
728  max_spin_ = agr::isFloatingPoint(max_) ? (agr::roundoff(max_ - min_, 1) * 10 + 1) : (max_ - min_ + 1);
729 
730  duration time_taken = steady_clock::now() - begin_timer;
731  float percentage_done = static_cast<float>(ticks_occurred) / (max_spin_);
732  duration time_left = time_taken * (1 / percentage_done - 1);
733  std::chrono::minutes minutes_left = std::chrono::duration_cast<std::chrono::minutes>(time_left);
734  std::chrono::seconds seconds_left =
735  std::chrono::duration_cast<std::chrono::seconds>(time_left - minutes_left);
736 
737  osm::cout << "["
738  << feat(sty, "italics") +
739  "Estimated time "
740  "left: " +
741  feat(rst, "italics")
742  << feat(col, "green") << minutes_left.count() << feat(rst, "color") << "m "
743  << feat(col, "green") << seconds_left.count() << feat(rst, "color") << "s"
