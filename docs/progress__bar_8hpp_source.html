691  // remaining_time
698  void remaining_time()
699  {
700  max_spin_ = agr::isFloatingPoint( max_ ) ?
701  ( agr::roundoff( max_ - min_, 1 ) * 10 + 1 ) :
702  ( max_ - min_ + 1 );
703 
704  duration time_taken = steady_clock::now() - begin_timer;
705  float percentage_done = static_cast <float> ( ticks_occurred ) / ( max_spin_ );
706  duration time_left = time_taken * ( 1 / percentage_done - 1 );
707  std::chrono::minutes minutes_left = std::chrono::duration_cast <std::chrono::minutes> ( time_left );
708  std::chrono::seconds seconds_left = std::chrono::duration_cast <std::chrono::seconds> ( time_left - minutes_left );
709 
710  std::cout << "["
711  << feat( sty, "italics" ) + "Estimated time left: " + feat( rst, "italics" )
712  << feat( col, "green" ) << minutes_left.count() << feat( rst, "color" ) << "m "
713  << feat( col, "green" ) << seconds_left.count() << feat( rst, "color" ) << "s"
714  << "]"
715  << feat( tcsc, "cln", 0 );
716  }
