512  iterating_var_ = 100 * (iterating_var - min_) /

522  output_ = feat(crs, "left", 100) + getColor() +

542  }

628  void remaining_time() {
629  max_spin_ = agr::isFloatingPoint(max_)
630  ? (agr::roundoff(max_ - min_, 1) * 10 + 1)
631  : (max_ - min_ + 1);
632 
633  duration time_taken = steady_clock::now() - begin_timer;
634  float percentage_done =
635  static_cast<float>(ticks_occurred) / (max_spin_);
636  duration time_left = time_taken * (1 / percentage_done - 1);
637  std::chrono::minutes minutes_left =
638  std::chrono::duration_cast<std::chrono::minutes>(time_left);
639  std::chrono::seconds seconds_left =
640  std::chrono::duration_cast<std::chrono::seconds>(time_left -
641  minutes_left);
642 
643  osm::cout << "["
644  << feat(sty, "italics") +
645  "Estimated time left: " + feat(rst, "italics")
646  << feat(col, "green") << minutes_left.count()
647  << feat(rst, "color") << "m " << feat(col, "green")
648  << seconds_left.count() << feat(rst, "color") << "s"
649  << "]" << feat(tcsc, "cln", 0);
